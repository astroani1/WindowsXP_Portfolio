        // --- SOUND ENGINE ---

        function playSound(type) {

            try {

                const sound = document.getElementById('sound-' + type);

                if (sound) {

                    sound.currentTime = 0;

                    sound.play().catch(e => console.log("Audio play failed (user interaction needed first):", e));

                }

            } catch (err) {

                console.log("Sound error:", err);

            }

        }

        // --- Logic for Clock ---

        function updateClock() {

            const now = new Date();

            let hours = now.getHours();

            const minutes = now.getMinutes();

            const ampm = hours >= 12 ? 'PM' : 'AM';

            hours = hours % 12;

            hours = hours ? hours : 12; 

            const strTime = hours + ':' + (minutes < 10 ? '0' + minutes : minutes) + ' ' + ampm;

            document.getElementById('clock').innerText = strTime;

        }

        setInterval(updateClock, 1000);

        updateClock();


        // --- Window Management ---

        let zIndexCounter = 100;

        let activeWindow = null;

        const openApps = new Set(); // Stores IDs of open windows


        function openWindow(id) {

            const win = document.getElementById(id);

            if (win.style.display !== 'flex') {

                playSound('click'); // Play click only on open

            }

            win.style.display = 'flex';

            openApps.add(id);

            bringToFront(id);

            if(id === 'paint') initPaint();

        }


        function closeWindow(id) {

            playSound('click');

            document.getElementById(id).style.display = 'none';

            openApps.delete(id);

            updateTaskbar();

        }


        function minimizeWindow(id) {

            playSound('click');

            document.getElementById(id).style.display = 'none';

            // Do NOT remove from openApps set when minimizing

            updateTaskbar();

        }

        

        function maximizeWindow(id) {

            playSound('click');

            const win = document.getElementById(id);

            if(win.style.width === '100%') {

                // Restore

                win.style.width = win.dataset.prevWidth || '500px';

                win.style.height = win.dataset.prevHeight || '400px';

                win.style.top = win.dataset.prevTop || '50px';

                win.style.left = win.dataset.prevLeft || '50px';

            } else {

                // Maximize

                win.dataset.prevWidth = win.style.width;

                win.dataset.prevHeight = win.style.height;

                win.dataset.prevTop = win.style.top;

                win.dataset.prevLeft = win.style.left;

                

                win.style.width = '100%';

                win.style.height = 'calc(100% - 30px)'; // minus taskbar

                win.style.top = '0';

                win.style.left = '0';

            }

        }


        function bringToFront(id) {

            zIndexCounter++;

            const win = document.getElementById(id);

            win.style.zIndex = zIndexCounter;

            

            // Highlight active window styling

            document.querySelectorAll('.window').forEach(w => w.classList.remove('active'));

            win.classList.add('active');

            

            activeWindow = id;

            updateTaskbar();

        }


        // --- Dragging Logic ---

        let isDragging = false;

        let currentWindow = null;

        let offset = { x: 0, y: 0 };


        function startDrag(e, id) {

            isDragging = true;

            currentWindow = document.getElementById(id);

            bringToFront(id);

            

            // Calculate offset based on mouse position inside the title bar

            const rect = currentWindow.getBoundingClientRect();

            offset.x = e.clientX - rect.left;

            offset.y = e.clientY - rect.top;

        }


        document.addEventListener('mousemove', (e) => {

            if (isDragging && currentWindow) {

                // Prevent dragging outside viewport roughly

                let x = e.clientX - offset.x;

                let y = e.clientY - offset.y;

                

                currentWindow.style.left = x + 'px';

                currentWindow.style.top = y + 'px';

            }

        });


        document.addEventListener('mouseup', () => {

            isDragging = false;

            currentWindow = null;

        });


        // --- Start Menu ---

        function toggleStartMenu() {

            const menu = document.getElementById('start-menu');

            if (menu.style.display === 'block') {

                menu.style.display = 'none';

            } else {

                menu.style.display = 'block';

                menu.style.zIndex = zIndexCounter + 2; // Always on top

            }

        }

        

        // Close start menu if clicked outside

        document.addEventListener('click', function(event) {

            const menu = document.getElementById('start-menu');

            const startBtn = document.getElementById('start-button');

            if (menu.style.display === 'block' && 

                !menu.contains(event.target) && 

                !startBtn.contains(event.target)) {

                menu.style.display = 'none';

            }

        });


        // --- Taskbar Logic ---

        

        function toggleWindowFromTaskbar(id) {

            playSound('click');

            const win = document.getElementById(id);

            if (win.style.display === 'none') {

                win.style.display = 'flex';

                bringToFront(id);

            } else {

                if (activeWindow === id) {

                    win.style.display = 'none'; // Minimize

                    updateTaskbar();

                } else {

                    bringToFront(id);

                }

            }

        }


        function updateTaskbar() {

            const container = document.getElementById('taskbar-items');

            container.innerHTML = '';

            

            openApps.forEach(id => {

                const win = document.getElementById(id);

                const title = win.querySelector('.title-bar-text span').innerText;

                

                const item = document.createElement('div');

                item.className = 'task-item';

                if (win.style.display !== 'none' && activeWindow === id) {

                    item.classList.add('active');

                }

                

                item.onclick = () => toggleWindowFromTaskbar(id);

                

                item.innerHTML = `<span>${title}</span>`;

                

                container.appendChild(item);

            });

        }

        

        // --- Paint Logic ---

        let paintCtx;

        let isPainting = false;

        let paintColor = '#000';

        let paintSize = 2;

        

        function initPaint() {

            const canvas = document.getElementById('paintCanvas');

            if(!paintCtx) {

                paintCtx = canvas.getContext('2d');

                paintCtx.fillStyle = '#fff';

                paintCtx.fillRect(0, 0, canvas.width, canvas.height);

                

                canvas.addEventListener('mousedown', startPainting);

                canvas.addEventListener('mousemove', draw);

                canvas.addEventListener('mouseup', stopPainting);

                canvas.addEventListener('mouseout', stopPainting);

            }

        }


        function startPainting(e) {

            isPainting = true;

            draw(e);

        }


        function stopPainting() {

            isPainting = false;

            paintCtx.beginPath();

        }


        function draw(e) {

            if (!isPainting) return;

            

            const canvas = document.getElementById('paintCanvas');

            const rect = canvas.getBoundingClientRect();

            const x = e.clientX - rect.left;

            const y = e.clientY - rect.top;


            paintCtx.lineWidth = paintSize;

            paintCtx.lineCap = 'round';

            paintCtx.strokeStyle = paintColor;


            paintCtx.lineTo(x, y);

            paintCtx.stroke();

            paintCtx.beginPath();

            paintCtx.moveTo(x, y);

        }


        function changeColor(color) {

            paintColor = color;

        }


        function changeSize(size) {

            paintSize = size;

        }


        function clearCanvas() {

            const canvas = document.getElementById('paintCanvas');

            paintCtx.fillStyle = '#fff';

            paintCtx.fillRect(0, 0, canvas.width, canvas.height);

        }
